```rust
//! Scripted demo-capture mode: drives the real editor window through a timed
//! step list and captures full-window frames via egui's viewport screenshot.
use std::path::PathBuf;

/// One captured frame: tightly packed RGBA8, row 0 at the top.
pub struct Frame {
    pub width: u32,
    pub height: u32,
    pub rgba: Vec<u8>,
}

/// Turns an egui screenshot reply into a `Frame` (unmultiplied RGBA8).
pub fn frame_from_color_image(img: &egui::ColorImage) -> Frame {
    let [w, h] = img.size;
    let mut rgba = Vec::with_capacity(w * h * 4);
    for px in img.pixels.iter() {
        let [r, g, b, a] = px.to_array();
        rgba.extend_from_slice(&[r, g, b, a]);
    }
    Frame { width: w as u32, height: h as u32, rgba }
}

/// True if the frame is not a single flat color (a blank/failed capture).
pub fn is_nonblank(frame: &Frame) -> bool {
    let n = frame.rgba.len();
    if n < 8 { return false; }
    let first = &frame.rgba[0..4];
    frame.rgba.chunks_exact(4).any(|px| px != first)
}
```
